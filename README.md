# fifo
SIGNAL WPTR RPTR: STD_LOGIC VECTOR(3 DOWNTO O);
TYPE FIFO IS ARRAY (15 DOWNTO OO)  OF STD_LOGIC_VECTOR(7 DOWNTO 0);
SIGNAL MEMO FIFO;

BEGIN
PROCESS(CLK,RST)
BEGIN

IF RST = '1' THEN
WPRT <= "0000";
RPRT <= "0000";
RED <= '0';

ELSEIF(CLK,EVENT AND CLK='1')THEN
IF(EN ='1')THEN 
IF(W='1')THEN
IF (WPTR < "1111") THEN
MEM(CONV_INTEGER(WPTR)) <= DATAIN;
WPTR <= WPTR +1;

ELSE 
RED <='1';
END IF;

ELSE
IF(RPTR < WPTR) THEN
RED <= '0';
DATAOUT <= MEM(CONV_INTEGER (RPTR));
RPTR <= RPTR+1;
DATAOUT <= "00000000";

END IF ;
END IF;
END IF;
END IF;
END PROCESS;
END BEHAVIORAL;
